Critique · 5 min
Base universelle CSS
Socle indispensable — à coller EN PREMIER dans le CSS
1.1
Reset box-sizing + overflow + images
style.css · ligne 1
Sans box-sizing:border-box, les paddings s'ajoutent aux largeurs déclarées et cassent les layouts. Sans overflow-x:hidden, un seul élément qui dépasse crée un scroll horizontal sur toute la page.
/* ================================================
   BASE — Coller absolument en premiere ligne du CSS
   ================================================ */
*, *::before, *::after {
  box-sizing: border-box;
  margin: 0;
  padding: 0;
}
html {
  overflow-x: hidden;
  scroll-behavior: smooth;
  -webkit-text-size-adjust: 100%; /* evite zoom auto iOS */
}
body {
  overflow-x: hidden;
  width: 100%;
  min-height: 100vh;
}
/* Toutes les images restent dans leur conteneur */
img, picture, video, svg, canvas, iframe {
  max-width: 100%;
  height: auto;
  display: block;
}

2.3
JavaScript du hamburger
script.js ou juste avant </body> 
// ===== HAMBURGER MENU =====
document.addEventListener('DOMContentLoaded', function () {
  const hamburger = document.querySelector('.hamburger');
  const navLinks  = document.querySelector('.nav-links');
  if (!hamburger || !navLinks) return;

  hamburger.addEventListener('click', function () {
    const isOpen = navLinks.classList.toggle('open');
    hamburger.classList.toggle('active', isOpen);
    hamburger.setAttribute('aria-expanded', isOpen);
    document.body.style.overflow = isOpen ? 'hidden' : '';
  });

  // Fermer au clic sur un lien
  navLinks.querySelectorAll('a').forEach(function (link) {
    link.addEventListener('click', function () {
      navLinks.classList.remove('open');
      hamburger.classList.remove('active');
      hamburger.setAttribute('aria-expanded', 'false');
      document.body.style.overflow = '';
    });
  });

  // Fermer si clic en dehors du menu
  document.addEventListener('click', function (e) {
    if (!hamburger.contains(e.target) && !navLinks.contains(e.target)) {
      navLinks.classList.remove('open');
      hamburger.classList.remove('active');
      hamburger.setAttribute('aria-expanded', 'false');
      document.body.style.overflow = '';
    }
  });

  // Fermer si redimensionnement > 900px
  window.addEventListener('resize', function () {
    if (window.innerWidth > 900) {
      navLinks.classList.remove('open');
      hamburger.classList.remove('active');
      hamburger.setAttribute('aria-expanded', 'false');
      document.body.style.overflow = '';
    }
  });
});

#hero, .hero {

  align-items: center;
  position: relative;

Typographie fluide
Tailles qui s'adaptent entre 320px et 1920px sans media query
6.1
Remplacer les tailles fixes par clamp()
style.css
clamp(min, fluide, max) : une seule règle remplace 3 breakpoints. La taille s'adapte automatiquement à la largeur d'écran.
h1 {
  font-size: clamp(1.6rem, 4vw + 0.8rem, 3.5rem);
  line-height: 1.12;
}
h2 {
  font-size: clamp(1.3rem, 3vw + 0.5rem, 2.5rem);
  line-height: 1.2;
}
h3 {
  font-size: clamp(1rem, 2vw + 0.4rem, 1.5rem);
  line-height: 1.3;
}
p, li {
  font-size: clamp(0.9rem, 1.5vw, 1.05rem);
  line-height: 1.7;
}
.section-label {
  /* Labels type "Nos partenaires", "Notre expertise" */
  font-size: clamp(0.68rem, 1.2vw, 0.82rem);
  letter-spacing: 0.1em;
  text-transform: uppercase;
